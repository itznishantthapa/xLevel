//========== Request Notification Permission ============
export const requestNotificationPermission = async () => {
  try {
    const app = getApp();
    const messaging = getMessaging(app);

    if (Platform.OS === 'android') {
      // Check Android version
      if (Platform.Version >= 33) {
        // Android 13+ requires POST_NOTIFICATIONS
        const granted = await PermissionsAndroid.request(
          PermissionsAndroid.PERMISSIONS.POST_NOTIFICATIONS
        );
        if (granted !== PermissionsAndroid.RESULTS.GRANTED) {
          console.log('Notification permission denied');
          return false;
        }
      }
      // Android < 13 → skip asking, just register
      await registerDeviceForRemoteMessages(messaging);
      return true;
    }

    // iOS → ask for permission
    const authStatus = await requestPermission(messaging);
    const enabled =
      authStatus === AuthorizationStatus.AUTHORIZED ||
      authStatus === AuthorizationStatus.PROVISIONAL;

    if (!enabled) {
      console.log('Notification permission denied (iOS)');
      return false;
    }

    // iOS auto-registers for remote notifications by default in RNFB unless disabled in firebase.json.
    // Avoid manual registration to prevent warning logs on iOS.
    return true;
  } catch (error) {
    console.log('Permission error:', error);
    return false;
  }

};

//========== Handle Navigation ============
const handleNotificationNavigation = (remoteMessage) => {
  console.log('🔷 LOG NAV 1: handleNotificationNavigation called');
  console.log('🔷 LOG NAV 1.1: Remote message:', JSON.stringify(remoteMessage, null, 2));
  const screen = remoteMessage?.data?.screen;
  if (screen) {
    console.log('🔷 LOG NAV 2: Navigating to screen:', screen, 'with data:', remoteMessage.data);
    NavigationService.navigate(screen, remoteMessage.data);
  } else if (remoteMessage?.notification) {
    console.log('🔷 LOG NAV 3: No screen specified, navigating to HomeTab');
    NavigationService.navigate('customerTabs', { screen: 'HomeTab' });
  } else {
    console.log('⚠️ LOG NAV 4: No navigation performed - no screen and no notification');
  }
};





//========== Setup Notification Listeners ============
export const setupNotificationListeners = async () => {
  try {
    const app = getApp();
    const messaging = getMessaging(app);
    
    console.log('🔵 LOG 1: Setting up notification listeners');
    // Setup Notifee foreground event handler
    const unsubscribeNotifee = notifee.onForegroundEvent(({ type, detail }) => {
      console.log('🔵 LOG 2: Notifee foreground event triggered - Type:', type);
      console.log('🔵 LOG 2.1: Event detail:', JSON.stringify(detail, null, 2));
      if (type === EventType.PRESS) {
        console.log('🔵 LOG 2.2: User pressed notification');
        const remoteMessage = {
          data: detail.notification?.data,
          notification: {
            title: detail.notification?.title,
            body: detail.notification?.body,
          },
        };
        console.log('🔵 LOG 2.3: Navigating with message:', JSON.stringify(remoteMessage, null, 2));
        handleNotificationNavigation(remoteMessage);
      }
    });

    // Foreground message handler
    const unsubscribeOnMessage = onMessage(messaging, async remoteMessage => {
      console.log('🟢 LOG 3: Foreground message received');
      console.log('🟢 LOG 3.1: Remote message:', JSON.stringify(remoteMessage, null, 2));
      const data = remoteMessage?.data;

      if (data) {
        console.log('🟢 LOG 3.2: Data exists in message');
        // Check for duplicate notifications using notif_id (track last 10 notifications)
        if (data.notif_id) {
          console.log('🟢 LOG 3.3: Checking notif_id:', data.notif_id);
          const storedIDs = await AsyncStorage.getItem('notif_ids_history');
          const notifHistory = storedIDs ? JSON.parse(storedIDs) : [];
          console.log('🟢 LOG 3.4: Stored notification IDs:', notifHistory);
          
          if (notifHistory.includes(data.notif_id)) {
            console.log('🔴 LOG 3.5: DUPLICATE DETECTED! Skipping notification:', data.notif_id);
            return;
          }
          
          // Add new ID and keep only last 10
          notifHistory.push(data.notif_id);
          if (notifHistory.length > 10) {
            notifHistory.shift(); // Remove oldest
          }
          
          console.log('🟢 LOG 3.6: New notification, storing notif_id:', data.notif_id);
          await AsyncStorage.setItem('notif_ids_history', JSON.stringify(notifHistory));
          await AsyncStorage.setItem('last_notif_id', data.notif_id); // Keep for compatibility
        } else {
          console.log('⚠️ LOG 3.3: NO notif_id in notification data!');
        }

        const title = data.title || remoteMessage?.notification?.title;
        const body = data.body || remoteMessage?.notification?.body;
        const largeIcon = data.largeIcon;
        const importance = data.importance || 'high';

        console.log('🟢 LOG 3.7: Displaying notification - Title:', title, 'Body:', body, 'Importance:', importance);

        // Map importance to channel ID
        const channelId = `${importance}_importance`;

        // Build android config
        const androidConfig = {
          channelId: channelId,
          smallIcon: 'ic_notification',
          pressAction: {
            id: 'default',
          },
        };

        // Only add largeIcon if it exists
        if (largeIcon) {
          androidConfig.largeIcon = largeIcon;
        }

        // Display notification using Notifee
        await notifee.displayNotification({
          title: title,
          body: body,
          data: data,
          android: androidConfig,
          ios: {
            foregroundPresentationOptions: {
              alert: true,
              badge: true,
              sound: importance === 'high',
            },
          },
        });
        console.log('🟢 LOG 3.8: Notification displayed successfully');
      } else {
        console.log('⚠️ LOG 3.9: No data in remote message');
      }
    });

    // Background notification opened handler
    const unsubscribeOnNotificationOpened = onNotificationOpenedApp(messaging, remoteMessage => {
      console.log('🟡 LOG 4: Background notification opened');
      console.log('🟡 LOG 4.1: Remote message:', JSON.stringify(remoteMessage, null, 2));
      handleNotificationNavigation(remoteMessage);
    });



    console.log('🟣 LOG 5: Checking for initial notification');
    const initialNotification = await getInitialNotification(messaging);
    console.log('🟣 LOG 5.1: Initial notification:', JSON.stringify(initialNotification, null, 2));

    // Fix: Ignore data-only notifications
    if (initialNotification && (initialNotification.notification?.title || initialNotification.notification?.body)) {
      console.log('🟣 LOG 5.2: Initial notification has title/body - navigating');
      handleNotificationNavigation(initialNotification);
    } else if (initialNotification) {
      console.log('🔴 LOG 5.3: Initial notification is DATA-ONLY - IGNORING to prevent duplicate');
    } else {
      console.log('🟣 LOG 5.4: No initial notification');
    }


    // Return cleanup function
    return () => {
      console.log('🔵 LOG 6: Cleaning up notification listeners');
      unsubscribeOnMessage();
      unsubscribeOnNotificationOpened();
      unsubscribeNotifee();
    };
  } catch (error) {
    console.log('🔴 LOG ERROR: Error in setupNotificationListeners:', error);
    if (__DEV__) console.log(error)
  }
};

export const setupNotificationChannel = async () => {
  if (Platform.OS === 'android') {
    // High importance channel - shows as popup/heads-up notification
    await notifee.createChannel({
      id: 'high_importance',
      name: 'High Importance Notifications',
      importance: AndroidImportance.HIGH,
      vibration: true,
      vibrationPattern: [250, 250, 250, 250],
      sound: 'default',
    });

    // Normal importance channel - shows in notification tray only
    await notifee.createChannel({
      id: 'normal_importance',
      name: 'Normal Notifications',
      importance: AndroidImportance.DEFAULT,
      vibration: true,
      vibrationPattern: [250, 250],
      sound: 'default',
    });

    // Low importance channel - shows silently in notification tray
    await notifee.createChannel({
      id: 'low_importance',
      name: 'Low Importance Notifications',
      importance: AndroidImportance.LOW,
      vibration: false,
    });

    // Min importance channel - minimal notification
    await notifee.createChannel({
      id: 'min_importance',
      name: 'Silent Notifications',
      importance: AndroidImportance.MIN,
      vibration: false,
    });
  }
};






//========== Post FCM Token ============
export const postFCMToken = async () => {
  try {
    const fcmToken = await AsyncStorage.getItem('@fcm_token');
    if (!fcmToken) {
      throw new Error('No authentication token or FCM token found');
    }
    const response = await API.post(endpoints.postFCMToken, { token: fcmToken });

    return response.data;
  } catch (error) {
    if (__DEV__) console.log(error)
  }
};

# # High importance - Shows as popup/heads-up notification
# send_push_notification(token, "Urgent!", "You won a prize!", importance="high")

# # Normal importance - Shows in notification tray with sound
# send_push_notification(token, "New message", "You have a new message", importance="normal")

# # Low importance - Shows silently in notification tray
# send_push_notification(token, "Info", "New content available", importance="low")

# # Min importance - Minimal notification, no sound/vibration
# send_push_notification(token, "Update", "App updated", importance="min")
def send_push_notification(token, title, body, data=None, importance="high", use_large_icon=True):
    try:
        if not token:
            return False, "FCM token is missing"

        if not data:
            data = {}

        # Add unique notification ID to prevent duplicates
        data["notif_id"] = str(int(time.time() * 1000))  # Timestamp in milliseconds

        # Add notification data to the data payload
        data["title"] = title
        data["body"] = body
        if use_large_icon:
            data["largeIcon"] = "https://play-lh.googleusercontent.com/3MGh1WwRV6dFkNHYX6cFIJ_WkQaowoxFbKe0P7x5AETHD1SpwseYLqLUs0lq2Up2GhqZWnY1UpgAzPZyvGTcOA=w480-h960-rw"
        data["importance"] = importance  # Add importance to data payload

        data = {k: str(v) for k, v in data.items()}

        normalized_importance = (importance or "high").lower()
        if normalized_importance not in {"high", "normal", "low", "min"}:
            normalized_importance = "high"

        android_priority = "high" if normalized_importance == "high" else "normal"

        message = messaging.Message(
            # Remove notification payload for Android - let Notifee handle display
            data=data,
            token=token,
            android=messaging.AndroidConfig(
                priority=android_priority,
            ),
            # For iOS, you still need the notification payload
            apns=messaging.APNSConfig(
                payload=messaging.APNSPayload(
                    aps=messaging.Aps(
                        alert=messaging.ApsAlert(
                            title=title,
                            body=body,
                        ),
                        sound="default" if normalized_importance == "high" else None,
                        badge=1,
                    )
                )
            ),
        )

        response = messaging.send(message)
        return True, response
    except Exception as e:
        return False, str(e)
